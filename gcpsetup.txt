# 1. Install prerequisites
sudo apt install apt-transport-https ca-certificates gnupg curl -y

# 2. Download and add the Google Cloud public key
curl -fsSL https://packages.cloud.google.com/apt/doc/apt-key.gpg | sudo gpg --dearmor -o /usr/share/keyrings/cloud.google.gpg

# 3. Add the Google Cloud SDK distribution URI as a package source
echo "deb [signed-by=/usr/share/keyrings/cloud.google.gpg] https://packages.cloud.google.com/apt cloud-sdk main" | sudo tee /etc/apt/sources.list.d/google-cloud-sdk.list

# 4. Update the package list and install the CLI
sudo apt update && sudo apt install google-cloud-cli -y

# 5. Verify the installation
gcloud --version
